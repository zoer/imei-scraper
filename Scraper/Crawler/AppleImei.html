def request_imei(imei)
  req = Net::HTTP::Post.new(URL, "User-Agent" => USER_AGENT)
  req.set_form_data(sn: imei, num: number)
  @res = Net::HTTP.start(URL.hostname, URL.port, use_ssl: true) do |http|
    http.request(req)
  end
end
